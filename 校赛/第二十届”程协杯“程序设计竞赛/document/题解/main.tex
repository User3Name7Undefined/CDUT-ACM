	\section{B}
	\begin{frame}{B.罚时}
		\begin{block}{分析}
			设最终解决问题的次序为a,b,c.....则总用时为$(t_a)+(t_a+t_b)+(t_a+t_b+t_c)+......$。
			
			先解决的题目会被计算更多次，因此需要先解决花费时间更少的题目。
		\end{block}
		
		\begin{block}{解法}
			把题目按照花费时间从小到大排序，设$T$为题目用时，则$T_i\leftarrow T_{i-1}+t_i$。
			
			求出$\sum_{i=1}^n T$即可。
		\end{block}
	\end{frame}
